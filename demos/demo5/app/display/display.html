                <html>
                <head>
                <script src="http://ajax.googleapis.com/ajax/libs/jquery/1.8.0/jquery.min.js"></script>
                <script>
                    function hideDisplayTitle() {
                        $('#displayTitle').hide(800);
                    }
            
                    function showDisplayTitle() {
                        $('#displayTitle').show(800);
                    }
            
                    function showVisibilityTitle() {
                        $('#visibilityTitle').css('visibility', 'visible');
                    }
            
                    function hideVisibilityTitle() {
                        $('#visibilityTitle').css('visibility', 'hidden');
                    }
                </script>
            </head>
            
            <body>
                <div class="div-line">
                    <h1>布局</h1>
                    <ol>
                        <li>
                            <h3>block & inline</h3>
                            <div style="border: 2px #000 solid;">这是div1</div>
                            <div style="border: 2px #000 solid;">这是div2</div>
                            <span style="border: 2px #000 solid;">这是span1</span> <span
                                style="border: 2px #000 solid;">这是span2</span>
                        </li>
                        <li>
                            <h3>更改display</h3>
                            <div style="border: 2px #000 solid; display: inline;">这是div1</div>
                            <div style="border: 2px #000 solid; display: inline;">这是div2</div>
                            <span style="border: 2px #000 solid;display: block;">这是span1</span> <span
                                style="border: 2px #000 solid; display: block;">这是span2</span>
                        </li>
                        <li>
                            <h3>display:none 与 visibility:hidden</h3>
                            <div>以下是display:none</div>
                            <h2 id="displayTitle" style="display: none;">标题</h2>
                            <div>以上是display:none</div>
                            <br>
                            <button onclick="showDisplayTitle()">Show</button>  
                            <button onclick="hideDisplayTitle()">hide</button>
                            <br>
                            <br>
                            <div>以下是visibility:hidden</div>
                            <h2 id="visibilityTitle" style="visibility:hidden">标题</h2>
                            <div>以上是visibility:hidden</div>
                            <br>
                            <button onclick="showVisibilityTitle()">Show</button>  
                            <button onclick="hideVisibilityTitle()">hide</button>
                        </li>
                    </ol>
            
                    </div>
