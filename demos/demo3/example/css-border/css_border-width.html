Code: 
                    <html>
                    <head>
                        <style>
                            p.one {
                            border-style: solid;
                            border-width: 3px;
                            width: 200px;
                            }

                            p.one-1 {
                            border-style: solid;
                            border-width: 5px 10px;
                            width: 200px;
                            }

                            p.one-2 {
                            border-style: solid;
                            border-width: 5px 20px 15px; /* 上边框、左右边框、下边框*/

                            }

                            p.one-3 {
                            border-style: solid;

                            width: 200px;
                            }

                            p.two {
                            border-left-style: solid;/*border-right-style border-top-style border-bottom-style*/
                            border-width: 5px;
                            width: 200px;
                            }

                            p.three {
                            border-left-style: solid;
                            border-top-style: solid;
                            border-width: 5px;
                            width: 200px;
                            }
