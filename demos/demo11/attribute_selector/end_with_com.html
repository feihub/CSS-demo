    <!DOCTYPE html>
    <html lang="en">
    <head>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width, initial-scale=1.0">
        <title>a[href&=".com"]</title>
        <style>
            /* 选择href的属性值以".com"结尾的每个a元素 将字体颜色设为红色 */
            a[href$=".com"] {
                color: red;
            }
        </style>
    </head>
    <body>
